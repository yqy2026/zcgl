## 🔧 验证规则

### 1. 基础验证
```typescript
const validationRules = {
  // 必填字段验证
  required: ['property_name', 'address', 'ownership_status', 'property_nature', 'usage_status', 'ownership_entity'],
  
  // 数值范围验证
  numberRange: {
    occupancy_rate: { min: 0, max: 100 },
    // 所有面积和金额字段都不能为负数
    positiveNumbers: ['total_area', 'usable_area', 'land_area', 'monthly_rent', 'deposit', 'annual_income', 'annual_expense']
  },
  
  // 字符串长度验证
  maxLength: {
    property_name: 200,
    address: 500,
    ownership_entity: 200,
    management_entity: 200,
    business_category: 100,
    // ... 其他字段
  }
};
```

### 2. 业务逻辑验证
```typescript
const businessValidation = {
  // 面积一致性验证
  areaConsistency: {
    'rented_area <= rentable_area': '已出租面积不能大于可出租面积',
    'usable_area <= total_area': '可使用面积不能大于总面积',
    'actual_property_area <= total_building_area': '实际房产面积不能大于证载建筑面积'
  },
  
  // 日期逻辑验证
  dateLogic: {
    'contract_end_date >= contract_start_date': '合同结束日期不能早于开始日期',
    'operation_agreement_end_date >= operation_agreement_start_date': '协议结束日期不能早于开始日期'
  },
  
  // 条件字段验证
  conditionalFields: {
    // 当使用状态为"出租"时，租户信息为必填
    'usage_status === "出租"': ['tenant_name', 'monthly_rent'],
    // 当是分租时，分租备注为必填
    'is_sublease === true': ['sublease_notes']
  }
};
```
